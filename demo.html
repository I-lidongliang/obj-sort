//假设有一个数组对象，根据id从小到大排序
        var list = [
            { name: "zhangsan", min_price: "49353"},
            { name: "bocai", min_price: "36927"},
            { name: "qingyi", min_price: "7481"},
            { name: "qingyi", min_price: "12707"},
            { name: "qingyi", min_price: "5283"}
        ];
        //定义一个函数
        function compare(property, dir){
            //默认升序
            dir = dir !== undefined? dir:0;
         return function(obj1,obj2){
             var value1 = obj1[property];
             var value2 = obj2[property];
             if (dir) {
                return value2 - value1;     // 降序
             }else{
                return value1 - value2;     // 升序
             };
         }
        }
        //使用方法
        list.sort(compare("min_price"));
        document.querySelector('#result').innerText = JSON.stringify(list, null, 4);
